def calculate_rectangle(): print("Rectangle Area and Perimeter Calculator") length = float(input("Enter the length of the rectangle: ")) width = float(input("Enter the width of the rectangle: ")) area = length * width perimeter = 2 * (length + width) print(f"Area of the rectangle: {area} square units") print(f"Perimeter of the rectangle: {perimeter} units") if __name__ == "__main__": calculate_rectangle()
